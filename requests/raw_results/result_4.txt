SQL> SELECT p.country_name, f.year, f.co2 *f.population_count AS CO2, SUM(f.co2 * f.population_count) OVER (ORDER BY f.year ROWS UNBOUNDED PRECEDING) AS CUMUL_CO2
  2  FROM COUNTRIES p, FACTS f, AGE_GROUPS a
  3  WHERE f.id_country = p.id_country AND f.id_age_group = a.id_age_group AND a.age_group = 'TOTAL' AND f.population_count IS NOT NULL AND f.co2 IS NOT NULL
  4  ORDER BY p.country_name, f.year;
